Create src/core/editorial-planner.js:
```javascript
import { loadStore } from '../services/store.js';
import { loadBrandKit } from '../core/brand-kit.js';
import { loadEditorialRules } from '../core/editorial-rules.js';
import { logger } from '../services/logger.js';

export async function generateEditorialCalendar(keywords, options = {}) {
  const brandKit = await loadBrandKit();
  const rules = await loadEditorialRules();
  const config = loadConfig();

  const {
    startDate = new Date(),
    weeksAhead = 4,
    articlesPerWeek = 1
  } = options;

  const calendar = [];
  const start = new Date(startDate);

  for (let w = 0; w < weeksAhead; w++) {
    for (let a = 0; a < articlesPerWeek; a++) {
      const kwIndex = w * articlesPerWeek + a;
      if (kwIndex >= keywords.length) break;

      const kw = keywords[kwIndex];
      const targetDate = new Date(start);
      targetDate.setDate(targetDate.getDate() + w * 7 + a);

      const title = generateTitle(kw, brandKit);
      const metaDescription = generateMetaDescription(kw, rules, brandKit);
      const outline = generateOutline(kw, rules);

      calendar.push({
        id: `article-${Date.now()}-${kwIndex}`,
        keyword: kw.word,
        funnelStage: kw.funnelStage,
        title,
        metaDescription,
        outline,
        targetDate: targetDate.toISOString().split('T')[0],
        priority: kw.score,
        status: 'planned'
      });
    }
  }

  // Save to store
  const store = await loadStore();
  await saveStore({
    ...store,
    articles: {
      ...(store.articles || {}),
      editorialCalendar: calendar,
      generatedAt: new Date().toISOString()
    }
  });

  logger.info(`[editorial-planner] Generated calendar with ${calendar.length} articles`);
  return calendar;
}

function generateTitle(keyword, brandKit) {
  const templates = {
    awareness: [
      `Guia Completo: {keyword} em 2024`,
      `Tudo sobre {keyword}: O Guia Definitivo`,
      `{keyword}: O Que Você Precisa Saber`
    ],
    consideration: [
      `{keyword} vs A Melhor Alternativa`,
      `Review Completo: {keyword}`,
      `As Melhores Opções de {keyword} em 2024`
    ],
    decision: [
      `Melhores {keyword} para Comprar em 2024`,
      `{keyword}: Comparativo de Preços`,
      `Como Escolher o Melhor {keyword}`
    ]
  };

  const templatesForStage = templates[keyword.funnelStage] || templates.awareness;
  const template = templatesForStage[Math.floor(Math.random() * templatesForStage.length)];

  return template.replace('{keyword}', capitalize(keyword.word));
}

function generateMetaDescription(keyword, rules, brandKit) {
  const maxLength = rules.metaDescriptionMaxLength || 160;
  const cta = brandKit.cta || 'Saiba mais';

  const base = `${capitalize(keyword.word)}: guia completo com dicas práticas. `;
  const withCta = base + cta;

  return withCta.length > maxLength ? withCta.substring(0, maxLength - 3) + '...' : withCta;
}

function generateOutline(keyword, rules) {
  const minSections = rules.minSections || 4;
  const sections = [
    { heading: 'Introdução', description: `O que é ${keyword.word} e por que é importante` },
    { heading: 'Benefícios', description: `Principais benefícios de ${keyword.word}` },
    { heading: 'Como Começar', description: `Guia passo a passo para iniciar com ${keyword.word}` },
    { heading: 'Erros Comuns', description: `Mistakes to avoid with ${keyword.word}` },
    { heading: 'Conclusão', description: `Resumo e próximos passos com ${keyword.word}` }
  ];

  return sections.slice(0, minSections);
}

function capitalize(str) {
  return str.charAt(0).toUpperCase() + str.slice(1);
}

function loadConfig() {
  try {
    const { config } = require('../config/index.js');
    return config;
  } catch {
    return {};
  }
}

import { saveStore } from '../services/store.js';
```

```javascript
import { runKeywordResearch } from '../../core/keyword-research.js';
import { generateEditorialCalendar } from '../../core/editorial-planner.js';
import { loadStore } from '../../services/store.js';

export default async function plan(args) {
  if (args.keywords || args.k) {
    const count = parseInt(args.count || args.n || '10', 10);

    console.log('Running keyword research...\n');
    const keywords = await runKeywordResearch();

    console.log(`\n=== Top ${Math.min(count, keywords.length)} Keyword Opportunities ===`);
    for (const kw of keywords.slice(0, count)) {
      console.log(`[${kw.funnelStage.toUpperCase()}] ${kw.word} (score: ${kw.score.toFixed(1)}, competitors: ${kw.competitorCount})`);
    }
  } else if (args.calendar || args.cal) {
    const store = await loadStore();
    const kwData = store.operations['keyword-research'];

    if (!kwData?.prioritized) {
      console.log('No keyword research found. Run "plan --keywords" first.');
      return;
    }

    const weeks = parseInt(args.weeks || '4', 10);
    const output = args.json || args.output === 'json';

    const calendar = await generateEditorialCalendar(kwData.prioritized, { weeksAhead: weeks });

    if (output) {
      console.log(JSON.stringify(calendar, null, 2));
    } else {
      console.log('\n=== Editorial Calendar ===');
      for (const article of calendar) {
        console.log(`\n${article.targetDate} [${article.funnelStage.toUpperCase()}]`);
        console.log(`  Keyword: ${article.keyword}`);
        console.log(`  Title: ${article.title}`);
        console.log(`  Priority: ${article.priority.toFixed(1)}`);
      }
    }
  } else {
    console.log('plan --keywords [--count N]  Research keywords and find gaps');
    console.log('plan --calendar [--weeks N]  Generate editorial calendar from keywords');
  }
}
```
