import { loadConfig } from '../config/index.js';
import { logger } from '../services/logger.js';
import { retryQueue } from '../services/retry-queue.js';

const ARVO_API_URL = process.env.ARVO_API_URL || 'https://api.arvo.ai/v1/generate';

export async function generateArticle(prompt, options = {}) {
  const config = loadConfig();
  const apiKey = config.ARVO_API_KEY;

  if (!apiKey) {
    throw new Error('ARVO_API_KEY not configured');
  }

  const { articleId, title, outline, keyword, funnelStage } = options;

  const systemPrompt = buildSystemPrompt(keyword, funnelStage);
  const userPrompt = buildUserPrompt(title, outline, keyword);

  if (process.env.DRY_RUN === 'true' || options.dryRun) {
    logger.info('[arvo] DRY RUN - would send to API');
    return {
      article: `DRY RUN ARTICLE for: ${title}\n\nKeyword: ${keyword}\n\nOutline:\n${outline.map(s => `- ${s.heading}: ${s.description}`).join('\n')}`,
      title,
      metaDescription: options.metaDescription || '',
      dryRun: true
    };
  }

  try {
    const result = await retryQueue.add(async () => {
      const response = await fetch(ARVO_API_URL, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          system_prompt: systemPrompt,
          user_prompt: userPrompt,
          max_tokens: options.maxTokens || 2000,
          temperature: options.temperature || 0.7
        })
      });

      if (!response.ok) {
        const err = await response.text();
        throw new Error(`Arvo API error: ${response.status} - ${err}`);
      }

      return response.json();
    });

    const article = result.article || result.content || result.text || '';
    const generatedTitle = result.title || title;
    const generatedMeta = result.meta_description || result.metaDescription || '';

    logger.info(`[arvo] Generated article: ${generatedTitle}`);

    return {
      article,
      title: generatedTitle,
      metaDescription: generatedMeta,
      articleId
    };
  } catch (err) {
    logger.error(`[arvo] Generation failed: ${err.message}`);
    throw err;
  }
}

function buildSystemPrompt(keyword, funnelStage) {
  return `You are an expert SEO content writer for a Brazilian Portuguese website.

Keyword focus: "${keyword}"
Funnel stage: ${funnelStage || 'awareness'}

Rules:
- Write in Brazilian Portuguese
- Use primary keyword in first 100 words
- Include keyword in title, H1, and at least 2 H2 subheadings
- Meta description: 140-160 characters with keyword
- Article length: 800-1500 words
- Use semi-formal tone, engaging but professional
- Include CTA at end matching the funnel stage
- No keyword stuffing - natural language only
- Structure: intro, body with H2/H3, conclusion with CTA`;
}

function buildUserPrompt(title, outline, keyword) {
  const outlineText = outline.map((s, i) =>
    `${i + 1}. ${s.heading}${s.description ? `: ${s.description}` : ''}`
  ).join('\n');

  return `Generate a complete SEO article based on this brief:

TITLE: ${title}
KEYWORD: ${keyword}
OUTLINE:
${outlineText}

Write the full article following the outline. Include an engaging introduction, substantive body sections with examples where appropriate, and a conclusion with a call-to-action.`;
}

export async function generateArticleFromCalendar(calendarEntry, options = {}) {
  return generateArticle(null, {
    articleId: calendarEntry.id,
    title: calendarEntry.title,
    outline: calendarEntry.outline,
    keyword: calendarEntry.keyword,
    funnelStage: calendarEntry.funnelStage,
    metaDescription: calendarEntry.metaDescription,
    ...options
  });
}
```

import { loadStore } from '../../services/store.js';
import { loadConfig } from '../../config/index.js';
import { generateArticleFromCalendar } from '../../integrations/arvo.js';
import { logger } from '../../services/logger.js';

export default async function generate(args) {
  const config = loadConfig();

  if (args.article || args.a) {
    const store = await loadStore();
    const calendar = store.articles?.editorialCalendar || [];

    if (calendar.length === 0) {
      console.log('No editorial calendar found. Run "plan --calendar" first.');
      return;
    }

    // Find article by id or use first pending
    let articleToGenerate = calendar.find(a => a.id === args.id);

    if (!articleToGenerate) {
      articleToGenerate = calendar.find(a => a.status === 'planned');
    }

    if (!articleToGenerate) {
      console.log('No pending articles found in calendar.');
      return;
    }

    console.log(`Generating article: ${articleToGenerate.title}`);
    console.log(`Keyword: ${articleToGenerate.keyword}\n`);

    // Show article brief for approval
    console.log('=== ARTICLE BRIEF ===');
    console.log(`Title: ${articleToGenerate.title}`);
    console.log(`Meta: ${articleToGenerate.metaDescription}`);
    console.log(`Outline:\n${articleToGenerate.outline.map(o => `  - ${o.heading}`).join('\n')}\n`);

    // Approval mode
    if (config.APPROVAL_MODE === 'true' && !args.yes && !args.y) {
      const readline = await import('readline');
      const rl = readline.createInterface({ input: process.stdin, output: process.stdout });
      const answer = await new Promise(resolve => {
        rl.question('Generate article with Arvo? (y/n): ', resolve);
      });
      rl.close();
      if (answer.toLowerCase() !== 'y') {
        console.log('Aborted.');
        return;
      }
    }

    const dryRun = args.dryRun || args.dry || process.env.DRY_RUN === 'true';

    if (dryRun) {
      console.log('\n=== DRY RUN - Would send to Arvo ===');
      const result = await generateArticleFromCalendar(articleToGenerate, { dryRun: true });
      console.log(result.article);
      return;
    }

    const result = await generateArticleFromCalendar(articleToGenerate);

    // Save generated article to store
    const updatedCalendar = calendar.map(a =>
      a.id === articleToGenerate.id
        ? { ...a, status: 'generated', generatedAt: new Date().toISOString(), article: result.article }
        : a
    );

    await saveStore({
      ...store,
      articles: { ...store.articles, editorialCalendar: updatedCalendar }
    });

    console.log('\n=== GENERATED ARTICLE ===');
    console.log(result.article);
    console.log(`\nSaved to store. Status: generated`);
  } else {
    console.log('generate --article [--id X] [--dry-run]  Generate article via Arvo');
  }
}

import { saveStore } from '../../services/store.js';
```
