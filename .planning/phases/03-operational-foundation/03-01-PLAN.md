**acceptance_criteria:**
- src/services/logger.js exports logger instance
- Logger writes to logs/seo-ranker-%DATE%.log
- Supports: info, warn, error, debug levels
- Console output with colors
- File rotation: max 7 files retained
- Environment variable LOG_LEVEL controls level

**action:**
Create src/services/logger.js:
```javascript
import winston from 'winston';
import { resolve } from 'path';
import { config } from '../config/index.js';

const logLevel = process.env.LOG_LEVEL || 'info';
const logDir = resolve(process.cwd(), config.LOGS_PATH || './logs');

const logFormat = winston.format.combine(
  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
  winston.format.errors({ stack: true }),
  winston.format.printf(({ timestamp, level, message, stack }) => {
    return `${timestamp} [${level.toUpperCase()}]: ${stack || message}`;
  })
);

const consoleFormat = winston.format.combine(
  winston.format.colorize(),
  winston.format.timestamp({ format: 'HH:mm:ss' }),
  winston.format.printf(({ timestamp, level, message, stack }) => {
    return `${timestamp} ${level}: ${stack || message}`;
  })
);

export const logger = winston.createLogger({
  level: logLevel,
  format: logFormat,
  transports: [
    new winston.transports.File({
      filename: resolve(logDir, 'seo-ranker-%DATE%.log'),
      datePattern: 'YYYY-MM-DD',
      maxFiles: 7,
      level: logLevel,
    }),
    new winston.transports.Console({
      format: consoleFormat,
    }),
  ],
});

export default logger;
```

### Task 2: Create context store with persistence

**read_first:**

**acceptance_criteria:**

- Context persisted to context/store.json
- Atomic writes (write to temp, then rename)
- Creates context/ directory if not exists

**action:**
Create src/services/store.js:
```javascript
import { readFile, writeFile, mkdir } from 'fs/promises';
import { resolve } from 'path';
import { randomUUID } from 'crypto';
import { config } from '../config/index.js';
import { logger } from './logger.js';

const storePath = resolve(process.cwd(), config.CONTEXT_PATH || './context', 'store.json');

async function ensureDir() {
  const dir = resolve(process.cwd(), config.CONTEXT_PATH || './context');
  try {
    await mkdir(dir, { recursive: true });
  } catch (e) {
    // already exists
  }
}

export async function loadStore() {
  try {
    const content = await readFile(storePath, 'utf-8');
    return JSON.parse(content);
  } catch (err) {
    if (err.code === 'ENOENT') {
      return { operations: {}, articles: {}, social: {} };
    }
    throw err;
  }
}

export async function saveStore(store) {
  await ensureDir();
  const tmpPath = storePath + '.tmp';
  await writeFile(tmpPath, JSON.stringify(store, null, 2), 'utf-8');
  // Atomic rename
  try {
    const { rename } = await import('fs/promises');
    await rename(tmpPath, storePath);
  } catch (err) {
    // Windows fallback
    await writeFile(storePath, JSON.stringify(store, null, 2), 'utf-8');
    await import('fs/promises').then(m => m.unlink(tmpPath).catch(() => {}));
  }
}

export async function getContext(key) {
  const store = await loadStore();
  return store.operations[key] || null;
}

export async function setContext(key, value) {
  const store = await loadStore();
  store.operations[key] = value;
  await saveStore(store);
}

export async function recordOperation(opId, data) {
  const store = await loadStore();
  if (!store.operations[opId]) {
    store.operations[opId] = { id: opId, status: 'pending', createdAt: new Date().toISOString() };
  }
  Object.assign(store.operations[opId], data);
  await saveStore(store);
}
```
