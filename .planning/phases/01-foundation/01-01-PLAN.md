- package.json exists with name "seo-ranker"
- package.json contains: node-fetch, xml2js, dotenv, chalk, node-schedule, winston, uuid
- `npm install` completes without error
- `node -e "require('./package.json')"` succeeds

**action:**
Create package.json:

{
  "name": "seo-ranker",
  "version": "1.0.0",
  "description": "SEO Content Marketing Automation - Autonomous agent",
  "type": "module",
  "main": "src/cli/index.js",
  "scripts": {
    "start": "node src/cli/index.js",
    "analyze": "node src/cli/index.js analyze",
    "plan": "node src/cli/index.js plan",
    "generate": "node src/cli/index.js generate",
    "publish": "node src/cli/index.js publish",
    "social": "node src/cli/index.js social",
    "run:weekly": "node src/cli/index.js run-weekly"

  "dependencies": {
    "chalk": "^5.3.0",
